Babai’s Nearest Plane algorithm

from: http://mslc.ctf.su/wp/plaidctf-2016-sexec-crypto-300/

def Babai_closest_vector(M, G, target):
small = target
for _ in range(1):
for i in reversed(range(M.nrows())):
c = ((small * G[i]) / (G[i] * G[i])).round()
small -= M[i] * c
return target - small
